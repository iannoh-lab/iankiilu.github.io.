<section id="mmu" class="card">
  <h2><i data-lucide="map"></i> MMU LiDAR Cartography Project</h2>
  <p>This interactive map visualizes simulated LiDAR sensor access points across Multimedia University of Kenya for activity monitoring and campus security. All coordinates are synthetic but geographically realistic for demonstration.</p>

<script>
lucide.createIcons();
var map = L.map('mmuMap').setView([-1.3570, 36.8070], 16);
L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
  attribution: '© OpenStreetMap contributors'
}).addTo(map);

var geojsonData = {
"type": "FeatureCollection",
"features": [
{"type":"Feature","properties":{"id":"MMU-L1","name":"Main Gate - North","model":"TF-Mini-S","coverage":30,"status":"active","notes":"Overlooks primary pedestrian flow"},"geometry":{"type":"Point","coordinates":[36.80685,-1.35672]}},
{"type":"Feature","properties":{"id":"MMU-L2","name":"Library Roof","model":"TF-Mini-S","coverage":40,"status":"active","notes":"High vantage point"},"geometry":{"type":"Point","coordinates":[36.80610,-1.35720]}},
{"type":"Feature","properties":{"id":"MMU-L3","name":"Admin Block East","model":"Velodyne VLP-16","coverage":60,"status":"active","notes":"Long-range coverage"},"geometry":{"type":"Point","coordinates":[36.80730,-1.35690]}},
{"type":"Feature","properties":{"id":"MMU-L4","name":"Sports Field NW","model":"TF-Mini-S","coverage":35,"status":"maintenance","notes":"Intermittent power"},"geometry":{"type":"Point","coordinates":[36.80540,-1.35830]}},
{"type":"Feature","properties":{"id":"MMU-L5","name":"Hostel Block A","model":"LD2410","coverage":25,"status":"active","notes":"Residence monitoring"},"geometry":{"type":"Point","coordinates":[36.80800,-1.35480]}}
]
};

geojsonData.features.forEach(function(f){
  var coords = f.geometry.coordinates;
  var radius = f.properties.coverage;
  var color = (f.properties.status === "active") ? "#2e7d32" : "#ff9800";

  var marker = L.circleMarker([coords[1], coords[0]], {
    radius: 6, fillColor: color, color: "#fff", weight: 1, fillOpacity: 1
  }).addTo(map);

  L.circle([coords[1], coords[0]], {
    color: color, fillColor: color, fillOpacity: 0.15, radius: radius
  }).addTo(map);

  marker.bindPopup("<b>" + f.properties.name + "</b><br>Model: " + f.properties.model + "<br>Status: " + f.properties.status + "<br>Coverage: " + radius + "m<br><i>" + f.properties.notes + "</i>");
});
